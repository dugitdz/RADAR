dt = diff(t);                     % diferenças entre timestamps
idx = find(dt > 0.5);             % índices onde dt supera 0.5s

if ~isempty(idx)
    fprintf("Quebra detectada em:\n");
    for k = 1:length(idx)
        fprintf("i = %d | t(i)=%.3f  →  t(i+1)=%.3f | dt=%.3f\n", ...
            idx(k), t(idx(k)), t(idx(k)+1), dt(idx(k)));
    end
else
    disp("Nenhum dt > 0.5 encontrado.");
end
